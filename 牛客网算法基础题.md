##### 问题

代码时间复杂度高：

- 外层循环：i从1000到98901（约9.8万次）
- 内层循环：j从10到sqrt(i)（约100-300次）
- 总循环次数：约1000万次以上

看起来不多，但每次循环还要做字符串操作，在牛客网测评环境容易超时

##### 优化方案

法1：减少循环范围+提前剪枝

```c++
#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

bool checkPandigital(string s) {
    if (s.length() != 9) return false;
    sort(s.begin(), s.end());
    return s == "123456789";
}

int main() {
    // 只考虑结果在1000-9876之间的四位数
    // 因为9个数字组成：结果+乘数+乘数，结果不可能是5位数
    for (int result = 1000; result <= 9876; result++) {
        // 只检查到sqrt(result)，避免重复
        for (int m1 = 1; m1 * m1 <= result; m1++) {
            if (result % m1 != 0) continue;
            int m2 = result / m1;
            
            // 快速剪枝：m1和m2的位数之和 + result的位数 = 9
            string s = to_string(result) + to_string(m1) + to_string(m2);
            if (s.length() == 9 && checkPandigital(s)) {
                cout << result << " = " << m1 << " x " << m2 << endl;
            }
        }
    }
    return 0;
}
```

法2：最优化-使用排列枚举

1. **最快**：只需枚举 362880 种排列，而不是近千万次循环
2. **最准确**：直接生成符合条件的数字组合，不会漏解
3. **最简洁**：代码逻辑清晰，易于理解

运行时间应该在 **0.1秒以内**，绝对不会超时。

既然题目要求1-9各出现一次，可以直接枚举数字的排列

```c++
#include <iostream>
#include <algorithm>
#include <string>
using namespace std;

int main() {
    string digits = "123456789";
    
    // 枚举1-9的所有排列（9! = 362880种）
    do {
        // 尝试不同的分割方式：结果(4位) × 乘数1(1-2位) × 乘数2(3-4位)
        // 因为总共9位，结果可能是3位或4位数
        
        // 情况1：结果4位，乘数1位×乘数4位 或 乘数2位×乘数3位
        for (int len1 = 1; len1 <= 4; len1++) {
            for (int len2 = 1; len2 <= 4; len2++) {
                int len3 = 9 - len1 - len2;
                if (len3 < 1 || len3 > 4) continue;
                if (len1 > len2) continue; // 保证较小的乘数在前
                
                string s1 = digits.substr(0, len1);
                string s2 = digits.substr(len1, len2);
                string s3 = digits.substr(len1 + len2, len3);
                
                int num1 = stoi(s1);
                int num2 = stoi(s2);
                int num3 = stoi(s3);
                
                if (num2 * num3 == num1) {
                    cout << num1 << " = " << num2 << " x " << num3 << endl;
                }
            }
        }
    } while (next_permutation(digits.begin(), digits.end()));
    
    return 0;
}
```

法3：针对性优化原来代码

如果保留原有思路，可以优化

```c++
#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;

int main() {
    ios::sync_with_stdio(false);  // 加速输入输出
    cin.tie(NULL);
    
    // 结果只可能是4位数（1000-9876）
    for (int i = 1000; i <= 9876; i++) {
        // 乘数范围：从1到sqrt(i)
        for (int j = 1; j * j <= i; j++) {  // 用乘法代替sqrt，更快
            if (i % j != 0) continue;
            int q = i / j;
            
            // 快速位数判断，避免不必要的字符串操作
            int totalDigits = 0;
            int temp = i;
            while (temp) { totalDigits++; temp /= 10; }
            temp = j;
            while (temp) { totalDigits++; temp /= 10; }
            temp = q;
            while (temp) { totalDigits++; temp /= 10; }
            if (totalDigits != 9) continue;
            
            // 拼接字符串
            string i1 = to_string(i);
            string j1 = to_string(j);
            string q1 = to_string(q);
            string m = i1 + j1 + q1;
            
            // 检查是否包含1-9各一次（用数组计数，比find更快）
            int cnt[10] = {0};
            for (char c : m) {
                if (c < '1' || c > '9') break;
                cnt[c - '0']++;
            }
            
            bool valid = true;
            for (int a = 1; a <= 9; a++) {
                if (cnt[a] != 1) {
                    valid = false;
                    break;
                }
            }
            
            if (valid) {
                cout << i << " = " << j << " x " << q << endl;
            }
        }
    }
    return 0;
}
```
