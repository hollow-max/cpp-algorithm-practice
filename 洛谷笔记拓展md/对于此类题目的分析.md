使用滑动窗口

```c++
#include <iostream>
#include <vector>
#include <unordered_map>
using namespace std;

int main() {
    int n, m, d;
    cin >> n >> m >> d;
    
    vector<vector<int>> a(n, vector<int>(m));
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            cin >> a[i][j];
        }
    }
    
    int maxArea = -1;
    
    // 枚举行的上边界
    for (int top = 0; top < n; top++) {
        // colFreq[j] 存储第 j 列的颜色计数
        vector<unordered_map<int, int>> colFreq(m);
        
        // 枚举行的下边界
        for (int bottom = top; bottom < n; bottom++) {
            // 将当前行的颜色加入对应列的统计中
            for (int j = 0; j < m; j++) {
                colFreq[j][a[bottom][j]]++;
            }
            
            // 现在用滑动窗口在列方向上找恰好 d 种颜色的区间
            unordered_map<int, int> windowFreq;  // 窗口内的颜色频率
            int distinct = 0;  // 窗口中不同颜色的数量
            int left = 0;
            
            for (int right = 0; right < m; right++) {
                // 将第 right 列的所有颜色加入窗口
                for (auto& p : colFreq[right]) {
                    int color = p.first;
                    int count = p.second;
                    if (windowFreq[color] == 0) {
                        distinct++;
                    }
                    windowFreq[color] += count;
                }
                
                // 如果颜色种类超过 d，移动左边界
                while (distinct > d && left <= right) {
                    for (auto& p : colFreq[left]) {
                        int color = p.first;
                        int count = p.second;
                        windowFreq[color] -= count;
                        if (windowFreq[color] == 0) {
                            distinct--;
                        }
                    }
                    left++;
                }
                
                // 如果恰好 d 种颜色，更新答案
                if (distinct == d) {
                    int height = bottom - top + 1;
                    int width = right - left + 1;
                    maxArea = max(maxArea, height * width);
                }
            }
        }
    }
    
    if (maxArea == -1) {
        cout << "Hrk" << endl;
    } else {
        cout << maxArea << endl;
    }
    
    return 0;
}
```

暴力枚举方法（n,m较小）

```c++
#include <iostream>
#include <vector>
#include <unordered_set>
using namespace std;

int main() {
    int n, m, d;
    cin >> n >> m >> d;
    
    vector<vector<int>> a(n, vector<int>(m));
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            cin >> a[i][j];
        }
    }
    
    int maxArea = -1;
    
    // 枚举所有子矩阵
    for (int top = 0; top < n; top++) {
        for (int left = 0; left < m; left++) {
            for (int bottom = top; bottom < n; bottom++) {
                for (int right = left; right < m; right++) {
                    unordered_set<int> colors;
                    // 统计颜色种类
                    for (int i = top; i <= bottom; i++) {
                        for (int j = left; j <= right; j++) {
                            colors.insert(a[i][j]);
                        }
                    }
                    if (colors.size() == d) {
                        int area = (bottom - top + 1) * (right - left + 1);
                        maxArea = max(maxArea, area);
                    }
                }
            }
        }
    }
    
    if (maxArea == -1) {
        cout << "Hrk" << endl;
    } else {
        cout << maxArea << endl;
    }
    
    return 0;
}
```

将元素插入到unordered_set(无序集合)中

`unordered_set` 的概念：

是c++标准库中的一种容器，特点：

1. 存储不重复的元素(自动去重)
2. 无序存储(元素顺序不重要)
3. 查找、插入、删除操作的平均时间复杂度为O(1)

`insert()` 方法的作用：

```c++
unordered_set<int> colors;  // 创建一个存储 int 类型的集合
colors.insert(5);   // 插入元素 5
colors.insert(3);   // 插入元素 3
colors.insert(5);   // 再次插入 5，但不会重复添加
```

在本题中

```c++
// 遍历子矩阵中的所有元素
for (int i = top; i <= bottom; i++) {
    for (int j = left; j <= right; j++) {
        colors.insert(a[i][j]);  // 将每个弹珠的颜色插入集合
    }
}
// 此时 colors.size() 就是子矩阵中不同颜色的数量
```

【对比数组方法】

若不用`unordered_set` ,用数组也可以实现：

```c++
// 方法1：使用数组标记
bool visited[401] = {false};  // 因为颜色最大 400
int distinct = 0;

for (int i = top; i <= bottom; i++) {
    for (int j = left; j <= right; j++) {
        int color = a[i][j];
        if (!visited[color]) {
            visited[color] = true;
            distinct++;
        }
    }
}
// distinct 就是不同颜色的数量
```

```c++
// 方法2：使用 unordered_set（更简洁）
unordered_set<int> colors;
for (int i = top; i <= bottom; i++) {
    for (int j = left; j <= right; j++) {
        colors.insert(a[i][j]);
    }
}
// colors.size() 就是不同颜色的数量
```

`unordered_set` 的常用操作：

```c++
#include <unordered_set>
using namespace std;

unordered_set<int> s;

s.insert(10);           // 插入元素
s.insert(20);
s.size();               // 返回元素个数：2
s.count(10);            // 检查是否存在：返回 1（存在）或 0（不存在）
s.find(20);             // 查找元素，返回迭代器
s.erase(10);            // 删除元素
s.empty();              // 判断是否为空
s.clear();              // 清空所有元素

// 遍历集合
for (int x : s) {
    cout << x << " ";
}
```

使用unordered_set的好处：

在本题中，我们需要统计子矩阵中**不同颜色**的数量，使用 `unordered_set` 的好处：

1. **自动去重**：不需要手动判断颜色是否已经出现过
2. **代码简洁**：一行代码就能完成插入
3. **高效**：插入和查找都是 O(1)
